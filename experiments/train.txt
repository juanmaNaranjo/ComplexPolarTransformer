    """
    Collate function para DataLoader que maneja batches de moléculas de distinto tamaño.
    Retorna un diccionario de tensores/lists.
    """
    return {
        "coords_cart": [item["coords_cart"] for item in batch],

    }

    dataset = QM9SDFDataset(

        csv_path=dataset_cfg["csv"],

    )

    y_values = [item["y"] for item in dataset]

    for item in dataset:
        item["y"] = (item["y"] - y_mean) / y_std

    # --- División train/val ---
    n_total = len(dataset)
    n_train = int(n_total * (1 - cfg["validation_split"]))

    train_ds, val_ds = random_split(dataset, [n_train, n_val])

    # --- DataLoaders ---
    train_dl = DataLoader(train_ds, batch_size=cfg["batch_size"], shuffle=True, collate_fn=collate_mol)
    val_dl = DataLoader(val_ds, batch_size=cfg["batch_size"], collate_fn=collate_mol)

    # --- Inicializar modelo ---
    model = ComplexPolarTransformerBeta(

        ckpt_dir="checkpoints"
    )

    # --- Entrenamiento ---
